import React, { useState, useMemo, useEffect } from 'react';

import { Plus, TrendingUp, Package, DollarSign, BarChart3, Eye, CheckCircle, Trophy, Star, Zap, Target, Award, Calendar } from 'lucide-react';


const MakeASale = () => {

  const [items, setItems] = useState([]);

  const [showQuickAdd, setShowQuickAdd] = useState(false);

  const [activeTab, setActiveTab] = useState('dashboard');

  const [reportPeriod, setReportPeriod] = useState('week');

  const [showCelebration, setShowCelebration] = useState(false);

  const [lastAction, setLastAction] = useState('');


  // Sale dialog state

  const [showSaleDialog, setShowSaleDialog] = useState(false);

  const [selectedItemId, setSelectedItemId] = useState(null);

  const [saleData, setSaleData] = useState({

    soldPrice: '',

    dateListed: '',

    dateSold: new Date().toISOString().split('T')[0]

  });


  // Quick add form state

  const [quickItem, setQuickItem] = useState({

    name: '',

    paidPrice: '',

    soldPrice: '',

    status: 'listed'

  });


  // Sound effects

  const playSound = (type) => {

    try {

      const audioContext = new (window.AudioContext || window.webkitAudioContext)();

      let frequency = 440;

      let duration = 200;

      

      switch(type) {

        case 'add':

          frequency = 523; // C note

          duration = 150;

          break;

        case 'sale':

          frequency = 659; // E note

          duration = 300;

          break;

        case 'levelup':

          // Play a chord

          [523, 659, 784].forEach((freq, i) => {

            setTimeout(() => playTone(audioContext, freq, 400), i * 100);

          });

          return;

        case 'achievement':

          frequency = 880; // High A

          duration = 500;

          break;

        default:

          frequency = 440;

      }

      

      playTone(audioContext, frequency, duration);

    } catch (e) {

      console.log('Audio not available');

    }

  };


  const playTone = (audioContext, frequency, duration) => {

    const oscillator = audioContext.createOscillator();

    const gainNode = audioContext.createGain();

    

    oscillator.connect(gainNode);

    gainNode.connect(audioContext.destination);

    

    oscillator.frequency.setValueAtTime(frequency, audioContext.currentTime);

    oscillator.type = 'sine';

    

    gainNode.gain.setValueAtTime(0.3, audioContext.currentTime);

    gainNode.gain.exponentialRampToValueAtTime(0.01, audioContext.currentTime + duration / 1000);

    

    oscillator.start(audioContext.currentTime);

    oscillator.stop(audioContext.currentTime + duration / 1000);

  };


  // Celebration effect

  const triggerCelebration = (action) => {

    setLastAction(action);

    setShowCelebration(true);

    setTimeout(() => setShowCelebration(false), 2000);

  };


  const openSaleDialog = (itemId) => {

    const item = items.find(i => i.id === itemId);

    setSelectedItemId(itemId);

    setSaleData({

      soldPrice: '',

      dateListed: item ? item.dateAdded : new Date().toISOString().split('T')[0],

      dateSold: new Date().toISOString().split('T')[0]

    });

    setShowSaleDialog(true);

  };


  const handleSaleSubmit = () => {

    if (!saleData.soldPrice || isNaN(saleData.soldPrice)) {

      alert('Please enter a valid sold price! 💰');

      return;

    }


    // Update the item's date listed if changed

    if (saleData.dateListed) {

      setItems(prev => prev.map(item => 

        item.id === selectedItemId 

          ? { ...item, dateAdded: saleData.dateListed }

          : item

      ));

    }


    const soldDateISO = new Date(saleData.dateSold + 'T12:00:00').toISOString();

    markAsSold(selectedItemId, saleData.soldPrice, soldDateISO);

    

    setShowSaleDialog(false);

    setSaleData({

      soldPrice: '',

      dateListed: '',

      dateSold: new Date().toISOString().split('T')[0]

    });

    setSelectedItemId(null);

  };


  const markAsSold = (id, soldPrice, dateSold) => {

    const oldGameStats = calculateGameStats(items);

    

    setItems(prev => prev.map(item => 

      item.id === id 

        ? { 

            ...item, 

            soldPrice: parseFloat(soldPrice) || 0, 

            status: 'sold', 

            dateUpdated: dateSold || new Date().toISOString()

          }

        : item

    ));

    

    // Check for level up after state update

    setTimeout(() => {

      playSound('sale');

      triggerCelebration('SOLD! +25 XP 💰🎉');

      

      // Check level up with updated items

      const updatedItems = items.map(item => 

        item.id === id 

          ? { 

              ...item, 

              soldPrice: parseFloat(soldPrice) || 0, 

              status: 'sold', 

              dateUpdated: dateSold || new Date().toISOString()

            }

          : item

      );

      

      const newGameStats = calculateGameStats(updatedItems);

      if (newGameStats.level > oldGameStats.level) {

        setTimeout(() => {

          playSound('levelup');

          triggerCelebration(`🎉 LEVEL UP! 🎉 Level ${newGameStats.level}! 👑`);

        }, 500);

      }

    }, 100);

  };


  // Gamification system

  const calculateGameStats = (items) => {

    const soldItems = items.filter(item => item.status === 'sold');

    const totalProfit = soldItems.reduce((sum, item) => sum + (item.soldPrice - item.paidPrice), 0);

    

    // Points system

    let points = 0;

    points += items.length * 10; // 10 points per item added

    points += soldItems.length * 25; // 25 points per sale

    points += Math.floor(totalProfit / 10) * 5; // 5 points per $10 profit

    

    // Bonus points for streaks and achievements

    const consecutiveSales = getConsecutiveSales(soldItems);

    points += consecutiveSales * 15; // Streak bonus

    

    // Level calculation (every 100 points = 1 level)

    const level = Math.floor(points / 100) + 1;

    const pointsToNextLevel = 100 - (points % 100);

    

    return {

      points,

      level,

      pointsToNextLevel,

      totalProfit,

      soldCount: soldItems.length,

      listedCount: items.filter(item => item.status === 'listed').length,

      streak: consecutiveSales

    };

  };


  const getConsecutiveSales = (soldItems) => {

    if (soldItems.length === 0) return 0;

    

    // Get unique dates of sales and sort them in descending order (most recent first)

    const saleDates = [...new Set(soldItems

      .map(item => new Date(item.dateUpdated || item.dateAdded).toDateString())

    )].sort((a, b) => new Date(b) - new Date(a));

    

    if (saleDates.length === 0) return 0;

    

    let streak = 0;

    const today = new Date();

    

    // Check if there was a sale today or yesterday to start the streak

    const mostRecentSale = new Date(saleDates[0]);

    const daysDiff = Math.floor((today - mostRecentSale) / (1000 * 60 * 60 * 24));

    

    // If the most recent sale is more than 1 day old, no current streak

    if (daysDiff > 1) return 0;

    

    // Count consecutive days with sales

    let currentDate = new Date(saleDates[0]);

    streak = 1; // Count the first sale day

    

    for (let i = 1; i < saleDates.length; i++) {

      const nextSaleDate = new Date(saleDates[i]);

      const daysBetween = Math.floor((currentDate - nextSaleDate) / (1000 * 60 * 60 * 24));

      

      // If the next sale is exactly 1 day before, continue the streak

      if (daysBetween === 1) {

        streak++;

        currentDate = nextSaleDate;

      } else {

        // Break in consecutive days, end streak counting

        break;

      }

    }

    

    return streak;

  };

      { name: "Deal Hunter 🔍", icon: "🔍", color: "from-blue-400 to-blue-600", description: "Finding sweet deals! 💎" },

      { name: "Profit King 👑", icon: "👑", color: "from-yellow-400 to-yellow-600", description: "Making bank consistently! 🏦" },

      { name: "Sales Wizard 🧙‍♂️", icon: "🧙‍♂️", color: "from-purple-400 to-purple-600", description: "Pure selling magic! ✨" },

      { name: "Flip Legend 🦄", icon: "🦄", color: "from-pink-400 to-pink-600", description: "Legendary status achieved! 🌟" },

      { name: "Sale God 🚀", icon: "🚀", color: "from-orange-400 to-red-600", description: "Ultimate reseller deity! ⚡👑" }

    ];

    

    const levelIndex = Math.min(level - 1, levels.length - 1);

    return levels[levelIndex];

  };

  // Date helpers

  const getWeekStart = (date) => {

    const d = new Date(date);

    const day = d.getDay();

    const diff = d.getDate() - day;

    return new Date(d.setDate(diff));

  };


  const getMonthStart = (date) => {

    return new Date(date.getFullYear(), date.getMonth(), 1);

  };


  // Analytics calculations

  const analytics = useMemo(() => {

    const now = new Date();

    const weekStart = getWeekStart(now);

    const monthStart = getMonthStart(now);


    const filterByPeriod = (items, period) => {

      if (period === 'lifetime') return items;

      const startDate = period === 'week' ? weekStart : monthStart;

      return items.filter(item => new Date(item.dateAdded) >= startDate);

    };


    const calculateStats = (filteredItems) => {

      const soldItems = filteredItems.filter(item => item.status === 'sold');

      const totalInvested = filteredItems.reduce((sum, item) => sum + item.paidPrice, 0);

      const totalRevenue = soldItems.reduce((sum, item) => sum + item.soldPrice, 0);

      const totalProfit = soldItems.reduce((sum, item) => sum + (item.soldPrice - item.paidPrice), 0);


      // Calculate average time to sell

      let avgTimeToSell = 0;

      if (soldItems.length > 0) {

        const totalDays = soldItems.reduce((sum, item) => {

          const dateAdded = new Date(item.dateAdded);

          const dateSold = new Date(item.dateUpdated || item.dateAdded);

          const daysDiff = Math.max(0, Math.floor((dateSold - dateAdded) / (1000 * 60 * 60 * 24)));

          return sum + daysDiff;

        }, 0);

        avgTimeToSell = totalDays / soldItems.length;

      }


      return {

        totalItems: filteredItems.length,

        listedItems: filteredItems.filter(item => item.status === 'listed').length,

        soldItems: soldItems.length,

        totalInvested,

        totalRevenue,

        totalProfit,

        avgProfit: soldItems.length > 0 ? totalProfit / soldItems.length : 0,

        sellThrough: filteredItems.length > 0 ? (soldItems.length / filteredItems.length) * 100 : 0,

        avgTimeToSell

      };

    };


    return {

      week: calculateStats(filterByPeriod(items, 'week')),

      month: calculateStats(filterByPeriod(items, 'month')),

      lifetime: calculateStats(filterByPeriod(items, 'lifetime'))

    };

  }, [items]);

              onClick={() => {

                setSaleData({

                  soldPrice: '',

                  dateListed: '',

                  dateSold: new Date().toISOString().split('T')[0]

                });

                setSelectedItemId(null);

              }}

              className="px-6 py-3 border-2 border-gray-300 text-gray-700 rounded-xl font-semibold hover:bg-gray-50 transition-colors"

              Cancel ❌

          </div>

      </div>

    );

  };

            {/* Locked Achievements Preview */}

            {achievements.length < 10 && (

      {showQuickAdd && <QuickAddForm />}
